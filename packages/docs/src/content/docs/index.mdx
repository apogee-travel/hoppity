import { Card, CardGrid } from "@astrojs/starlight/components";

<CardGrid>
    <Card icon="puzzle" title="Contract-Driven Topology">
        Define domain contracts with Zod schemas. Hoppity derives all exchanges, queues, bindings,
        publications, and subscriptions automatically. No topology files. No manual Rascal config.
    </Card>
    <Card icon="setting" title="Builder Pattern">
        Fluent API with `hoppity.service(name, config).use(middleware).build()`. Declare handlers
        and publishes, get back a fully-wired `ServiceBroker`.
    </Card>
    <Card icon="seti:typescript" title="TypeScript-First">
        Written in TypeScript with full type safety. Strict mode, proper generics, Zod-inferred
        payload types from contract to handler to broker method.
    </Card>
    <Card icon="rocket" title="Built on Rascal">
        Standing on the shoulders of a proven RabbitMQ library. Hoppity adds contract-driven
        topology derivation, typed handlers, and composable middleware on top.
    </Card>
    <Card icon="random" title="Built-in RPC">
        Request/reply with correlation IDs, configurable timeouts, automatic reply queue management,
        and typed request/response contracts — all in core.
    </Card>
    <Card icon="list-format" title="Middleware for Cross-Cutting Concerns">
        Custom logging via `hoppity-logger`, or write your own middleware for any cross-cutting
        concern. Delayed delivery is built into core via contract-level `delay` declarations.
    </Card>
    <Card icon="bars" title="Interceptors and Observability">
        Wrap handler and publish execution for tracing, metrics, and header injection.
        `hoppity-open-telemetry` ships ready-made OTel interceptors — drop in `withTracing` and
        `withMetrics` and you are done.
    </Card>
</CardGrid>
